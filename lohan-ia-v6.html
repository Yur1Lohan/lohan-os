from openai import OpenAI
import sounddevice as sd
from scipy.io.wavfile import write
import tempfile, os
from dotenv import load_dotenv

load_dotenv()
client = OpenAI()

def gravar_audio():
    fs, seg = 44100, 5
    print("Fale agora...")
    audio = sd.rec(int(seg * fs), samplerate=fs, channels=1)
    sd.wait()
    arq = tempfile.NamedTemporaryFile(delete=False, suffix=".wav")
    write(arq.name, fs, audio)
    return arq.name

def voz_para_texto(path):
    return client.audio.transcriptions.create(
        model="gpt-4o-transcribe", file=open(path, "rb")).text

def gerar_resposta(texto):
    return client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role":"user","content":texto}]
    ).choices[0].message.content

def texto_para_voz(texto):
    r = client.audio.speech.create(
        model="gpt-4o-mini-tts", voice="alloy", input=texto)
    arq = "resposta.mp3"
    r.stream_to_file(arq)
    os.system(f"start {arq}")   # Windows
    # os.system(f"afplay {arq}") # Mac
    # os.system(f"mpg123 {arq}") # Linux

while True:
    audio = gravar_audio()
    pergunta = voz_para_texto(audio)
    print(f"Você: {pergunta}")
    resposta = gerar_resposta(pergunta)
    print(f"IA: {resposta}")
    texto_para_voz(resposta)
